\section{Linear buckling}

\begin{exampledeck}
*LOADCASE, TYPE=LINEARBUCKLING
  *SUPPORTS
  FIX
  *LOADS
  PRELOAD
  *NUMEIGENVALUES
  6
  *SIGMA
  0.0
  *REQUESTSTGEOM, FILE=buckling_geom
\end{exampledeck}

\section{Transient load with amplitude}

\begin{exampledeck}
*AMPLITUDE, NAME=RAMP, TYPE=LINEAR
0.0, 0.0
1.0, 1.0

*CLOAD, LOAD_COLLECTOR=F_RAMP, AMPLITUDE=RAMP
TIP, 0, 0, -1000, 0, 0, 0

*LOADCASE, TYPE=LINEARTRANSIENT
  *SUPPORTS
  FIX
  *LOADS
  F_RAMP
  *TIME
  0.0, 1.0, 0.01
  *NEWMARK
  0.25, 0.5
  *WRITEEVERY, TYPE=STEPS
  10
\end{exampledeck}
